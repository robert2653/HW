#include <stdio.h>
#include <stdlib.h>

int main()
{
    int t,n,k,judge=100,i,compare=100;

    {

        judge = 100;

            scanf("%d",&a[i]);

                    judge = 1;

                    compare=judge;
                }

                {
                    judge = 0;
                }

                {
                    judge = 1;
                }

                    judge = 2;
                }

                    compare = judge;
                }

                    judge = 1;
                }

                    judge = 2;

                {
                    judge = 3;

                {
                    judge = 4;
                }
                if(compare>judge)
                {
                    compare = judge;
                }
            }printf("%d\n",compare);
        }


        if(k==4)
        {

                        judge = 2;

                        judge = 1;

                     if(compare>judge)

                        compare = judge;

                if(n>=2)
                {
                    if(a[i]%4==0)
                    {
                        even=even+2;

                    {
                        even++;
                    }

                    if(even>=2)
                    {
                        judge = 0;
                    }
                    if(even==1)
                    {
                        judge = 1;
                    }
                    if(even==0)

                        if(a[i]%4==3)
                        {
                            judge=1;

                }
            }printf("%d\n",compare);

        }


    }

}
