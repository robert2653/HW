#include <stdio.h>
#include <stdlib.h>



int Prime(int n)
{
    if(n<=1)return 0;
    for(int i=2;i*i<=n;i++)
    {
        if(n%i == 0)
        {
            return 0;
        }
    }
    return 1;
}



int main()
{
     long int t;
    scanf("%d",&t);

    long long int num[t];
    for(int i =0;i<t;i++)
    {
        scanf("%d",&num[i]);
    }

    for(int i =0;i<t;i++)
    {
        if(Prime(num[i]))
         {
             printf("Yes\n");
         }else { printf("No\n");}
    }

   return 0;
}
