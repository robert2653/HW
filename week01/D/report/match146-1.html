#include <stdio.h>

int main(){
    long long int a1,a2,b1,b2,c1,c2,ab1,ab2,ac1,ac2,s;
    scanf("%lld %lld %lld %lld %lld %lld",&a1,&a2,&b1,&b2,&c1,&c2);
    ab1=b1-a1;


    ab2=b2-a2;
    ac1=c1-a1;

    s=ab1*ac2-ab2*ac1;
